# FLASK-TUTORIAL

# Prerequisites
Creating a virtual environment

    pip install virtualenv


# Requirements.txt

    pip3 install -r requirements.txt
    
#  Steps to create a flask application 

    py -3 venv myenv
    myenv\scripts\activate
    
# Running the application

    python main.py
    
# Viewing the App

    http://127.0.0.1:5000
